- 哲学家进餐问题

```C
semaphore fork[5] = {1};
int i;
void philosopher(int i)
{
  while (true) {
    think();
    P(fork[i]);
    P(fork[(i + 1) % 5]);
    eat();
    V(fork[(i + 1) % 5]);
    V(fork[i]);
  }
}

void main()
{
  multiProcess(philosopher(1), philosopher(2), philosopher(3), philosopher(4), philosopher(5));
}
```
